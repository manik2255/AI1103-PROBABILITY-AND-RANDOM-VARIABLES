\section{problem 6.14}
Events $A$ and $B$ are such that $\pr{A} = \frac{1}{2}$, $\pr{B} = \frac{7}{12}$ and $\pr{not\: A \:or\: not\: B} = \frac{1}{4}$. State whether $A$ and $B$ are independent ?
\section{solution}
\begin{align}
\pr{not\: A \:or\: not\: B} = \pr{A' + B'} 
\end{align}
We know that,\\
\begin{align} \label{eq:1}
 \pr{A' + B'} = \pr{(AB)'}
\end{align}
As,\\

\begin{align}
(AB)(AB)' = 0 \\
\pr{AB} + \pr{(AB)'} = 1
\end{align}
\begin{align} \label{eq:2}
\pr{AB} = 1- \pr{(AB)'}
\end{align}
Using ~\ref{eq:1} in ~\ref{eq:2}, We get
\begin{align} \label{eq:3}
\pr{AB} = 1- \pr{A' + B'}
\end{align}
On substituting the value of $\pr{A' + B'}$ in ~\ref{eq:3}, we get
\begin{align}
\pr{AB} = 1- \frac{1}{4}
\end{align}
\begin{align} \label{eq:4}
 \implies \pr{AB} = \frac{3}{4}
\end{align}
Given,
$\pr{A} = \frac{1}{2} \:and \:\pr{B} = \frac{7}{12}$
\begin{align} \label{eq:5}
\implies \pr{A}\pr{B} = \frac{7}{24}
\end{align}
From ~\ref{eq:4} and ~\ref{eq:5},
\begin{align}
 \pr{AB}\neq \pr{A}\pr{B}
\end{align}
As the events $A$ and $B$ does not satisfy the definition of independent events,\\
$\therefore$ Events $A$ and $B$ are dependent.
